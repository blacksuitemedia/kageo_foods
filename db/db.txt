-- 1. Create the blogs table [cite: 5]
CREATE TABLE IF NOT EXISTS blogs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    author VARCHAR(100),
    content TEXT,
    image_path VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. Update products table with new columns [cite: 1, 2, 4]
ALTER TABLE products 
ADD COLUMN ingredients TEXT,
ADD COLUMN nutrition_info TEXT,
ADD COLUMN status TINYINT(1) DEFAULT 1;

-- 3. Update categories table [cite: 6]
ALTER TABLE categories 
ADD COLUMN status TINYINT(1) DEFAULT 1;

-- 4. Ensure blogs table has necessary columns if it was already created 
-- (This runs just in case the table existed before the CREATE TABLE script) [cite: 3]
-- Note: If the table was created by the command above, these lines are redundant
-- but safe to keep.
ALTER TABLE blogs 
ADD COLUMN IF NOT EXISTS image_path VARCHAR(255) AFTER title,
ADD COLUMN IF NOT EXISTS author VARCHAR(100) AFTER image_path;
